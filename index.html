What is Unime?

Unime is an AI created to replicate the personality of unipcs using language modelling and X API.

It monitors unipcs tweets and responds in his style.

The AI learns from unipcs' communication patterns and automatically generates responses that match his personality.

Technical Implementation

Complete X API integration for monitoring and personality replication:

1. Setting up X API Connection

// Initialize X API client with credentials const { TwitterApi } = require('twitter-api-v2'); const OpenAI = require('openai'); const twitterClient = new TwitterApi({ appKey: process.env.TWITTER_API_KEY, appSecret: process.env.TWITTER_API_SECRET, accessToken: process.env.TWITTER_ACCESS_TOKEN, accessSecret: process.env.TWITTER_ACCESS_SECRET, }); const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

2. Real-time Tweet Monitoring

// Set up streaming rules to monitor unipcs async function setupMonitoring() { const rules = [ { value: 'from:unipcs', tag: 'unipcs_tweets' }, { value: '@unipcs', tag: 'unipcs_mentions' } ]; // Delete existing rules const existingRules = await twitterClient.v2.streamRules(); if (existingRules.data?.length) { await twitterClient.v2.updateStreamRules({ delete: { ids: existingRules.data.map(rule => rule.id) } }); } // Add new rules await twitterClient.v2.updateStreamRules({ add: rules }); // Start streaming const stream = await twitterClient.v2.searchStream({ 'tweet.fields': ['author_id', 'created_at', 'context_annotations'], 'user.fields': ['username'] }); stream.on('data', handleTweet); }

3. Personality-Based Response Generation

// Generate response that matches unipcs personality async function generatePersonalityResponse(originalTweet, context) { const personalityPrompt = ` You are unipcs, replying to this tweet: "${originalTweet.text}" Your personality traits: - Concise and direct communication style - Tech-focused discussions - Casual but knowledgeable tone - Often uses crypto/tech terminology - Replies are usually short (under 100 characters) Generate a response that sounds exactly like unipcs would respond: `; const completion = await openai.chat.completions.create({ model: "gpt-4", messages: [ { role: "system", content: personalityPrompt }, { role: "user", content: originalTweet.text } ], max_tokens: 50, temperature: 0.8 }); return completion.choices[0].message.content.trim(); }

4. Tweet Processing and Response

// Handle incoming tweets and generate responses async function handleTweet(tweet) { try { console.log('New tweet detected:', tweet.text); // Skip retweets and replies to avoid spam if (tweet.text.startsWith('RT ') || tweet.text.startsWith('@')) { return; } // Add delay to seem more natural const delay = Math.random() * 30000 + 10000; // 10-40 seconds setTimeout(async () => { const response = await generatePersonalityResponse(tweet); // Post the reply await twitterClient.v2.reply(response, tweet.id); console.log('Replied:', response); }, delay); } catch (error) { console.error('Error processing tweet:', error); } }

5. Rate Limiting and Error Handling

// Implement rate limiting to avoid API limits class RateLimiter { constructor(maxRequests, timeWindow) { this.maxRequests = maxRequests; this.timeWindow = timeWindow; this.requests = []; } async executeRequest(requestFn) { const now = Date.now(); this.requests = this.requests.filter(time => now - time < this.timeWindow ); if (this.requests.length >= this.maxRequests) { const waitTime = this.timeWindow - (now - this.requests[0]); await new Promise(resolve => setTimeout(resolve, waitTime)); } this.requests.push(now); return await requestFn(); } } const rateLimiter = new RateLimiter(50, 15 * 60 * 1000); // 50 requests per 15 minutes
